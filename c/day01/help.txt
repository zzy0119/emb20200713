
如果你写程序的时候忘记了其中用到语句含义？请看！！！！

运算符
	+ - * /(整除) %(求余)
	a += b 等效于 a = a+b
	&& 逻辑与，也就是和 a && b只有a非零并且b非零结果才为真
	|| 逻辑或，也就是或者 a || b 只要a和b中一个非零结果就非零
	! 逻辑取反	!a 如果a为0结果为真 如果a为非零结果为假
	?: 条件运算符，相当于if...else  a > b ? a : b表达含义为如果a>b为真则结果为a，否则为b
	++ 自增运算符 a++ 相当与a = a+1
	-- 自减运算符 a-- 相当于a = a-1

条件语句
	if (表达式) {
		// 如果表达式为真，则执行以下语句
	}

	也可
	if (表达式) {
	
	} else {
		// 表达式为假，则执行一下语句
	}
	
	多条件时, 有多少个条件就有多少个else if
	if (表达式) {
	
	} else if(表达式) {
	
	} else if (表达式2) {
	
	} else {
	
	}

	多条件时,还可以
	switch(常量表达式) {
		case 值1:
			break;
		case 值2:
			break;	
		case 值3:
			break;
		default:
			break;
	}

循环语句
	for (循环变量初始化; 循环条件; 循环变量的改变) {
		循环体,则循环语句
	}

	while (循环条件) {
		循环体	
	}

	do {
		循环体
	} while(循环条件);

	关键字:break;跳出所在循环
		continue;结束本次循环，继续下一次
